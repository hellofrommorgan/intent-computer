### TASK 3: Make sync-skills.js portable

File: scripts/sync-skills.js

Line 22 has a hardcoded path:
```javascript

```

Replace with a portable resolution using environment variable or CLI argument:

```javascript
// Resolve plugin root from CLI argument, environment variable, or default
const PLUGIN_ROOT = process.argv[2]
  || process.env.ARSCONTEXTA_PLUGIN_ROOT
  || (() => {
    const home = process.env.HOME || process.env.USERPROFILE || "";
    // Try to find the latest version in the plugin cache
    const cacheDir = path.join(home, ".claude", "plugins", "cache", "agenticnotetaking", "arscontexta");
    try {
      const versions = fs.readdirSync(cacheDir).sort();
      if (versions.length > 0) {
        return path.join(cacheDir, versions[versions.length - 1]);
      }
    } catch { /* fall through */ }
    // Hardcoded fallback for development
    return path.join(home, ".claude", "plugins", "cache", "agenticnotetaking", "arscontexta", "0.8.0");
  })();
```
